1. Star Pattern Generator

This program creates a beautiful star pattern using nested while loops.

#include <stdio.h>

int main() {
    int rows, i = 1, j;
    
    printf("Enter number of rows: ");
    scanf("%d", &rows);
    
    while (i <= rows) {
        j = 1;
        while (j <= i) {
            printf("* ");
            j++;
        }
        printf("\n");
        i++;
    }
    
    return 0;
}

2. Prime Number Checker

This program checks if a number is prime using a while loop.

#include <stdio.h>

int main() {
    int num, i = 2, isPrime = 1;
    
    printf("Enter a positive integer: ");
    scanf("%d", &num);
    
    if (num <= 1) {
        isPrime = 0;
    } else {
        while (i <= num / 2) {
            if (num % i == 0) {
                isPrime = 0;
                break;
            }
            i++;
        }
    }
    
    if (isPrime)
        printf("%d is a prime number.\n", num);
    else
        printf("%d is not a prime number.\n", num);
    
    return 0;
}

3. Armstrong Number Checker

This program checks if a number is an Armstrong number using while loops.

#include <stdio.h>
#include <math.h>

int main() {
    int num, original, remainder, digits = 0, result = 0;
    
    printf("Enter an integer: ");
    scanf("%d", &num);
    
    original = num;
    
    // Count digits
    while (original != 0) {
        original /= 10;
        digits++;
    }
    
    original = num;
    
    // Calculate sum of nth power of digits
    while (original != 0) {
        remainder = original % 10;
        result += pow(remainder, digits);
        original /= 10;
    }
    
    if (result == num)
        printf("%d is an Armstrong number.\n", num);
    else
        printf("%d is not an Armstrong number.\n", num);
    
    return 0;
}

4. GCD Calculator

This program calculates the Greatest Common Divisor using a while loop.

#include <stdio.h>

int main() {
    int num1, num2, gcd;
    
    printf("Enter two positive integers: ");
    scanf("%d %d", &num1, &num2);
    
    // Find GCD
    while (num1 != num2) {
        if (num1 > num2)
            num1 -= num2;
        else
            num2 -= num1;
    }
    
    gcd = num1;
    
    printf("GCD = %d\n", gcd);
    
    return 0;
}

2. Factorial Calculator

This program calculates the factorial of a number using a do-while loop.

#include <stdio.h>

int main() {
    int n, i = 1;
    unsigned long long factorial = 1;
    
    printf("Enter a positive integer: ");
    scanf("%d", &n);
    
    if (n < 0) {
        printf("Error! Factorial of a negative number doesn't exist.\n");
    } else {
        do {
            factorial *= i;
            i++;
        } while (i <= n);
        
        printf("Factorial of %d = %llu\n", n, factorial);
    }
    
    return 0;
}

3. Perfect Number Checker

This program checks if a number is a perfect number using a do-while loop.

#include <stdio.h>

int main() {
    int num, i = 1, sum = 0;
    
    printf("Enter a positive integer: ");
    scanf("%d", &num);
    
    do {
        if (num % i == 0) {
            sum += i;
        }
        i++;
    } while (i <= num / 2);
    
    if (sum == num && num != 0)
        printf("%d is a perfect number.\n", num);
    else
        printf("%d is not a perfect number.\n", num);
    
    return 0;
}

4. Binary Converter

This program converts a decimal number to binary using a do-while loop.

#include <stdio.h>

int main() {
    int num, binary[32], i = 0;
    
    printf("Enter a decimal number: ");
    scanf("%d", &num);
    
    printf("Binary equivalent of %d is: ", num);
    
    do {
        binary[i] = num % 2;
        num = num / 2;
        i++;
    } while (num > 0);
    
    // Print binary in reverse order
    do {
        i--;
        printf("%d", binary[i]);
    } while (i > 0);
    
    printf("\n");
    
    return 0;
}

5. Palindrome Checker

This program checks if a number is a palindrome using a do-while loop.

#include <stdio.h>

int main() {
    int num, reversed = 0, remainder, original;
    
    printf("Enter an integer: ");
    scanf("%d", &num);
    
    original = num;
    
    // Reverse the number
    do {
        remainder = num % 10;
        reversed = reversed * 10 + remainder;
        num /= 10;
    } while (num != 0);
    
    // Check palindrome
    if (original == reversed)
        printf("%d is a palindrome.\n", original);
    else
        printf("%d is not a palindrome.\n", original);
    
    return 0;
}

While vs. Do-While: What's the Difference?

While Loop

  • Condition is checked before execution
  • May execute zero or more times
  • Use when iteration count is unknown
  • Syntax: while (condition) { ... }

Do-While Loop

  • Condition is checked after execution
  • Executes at least once
  • Use when you need guaranteed execution
  • Syntax: do { ... } while (condition);
